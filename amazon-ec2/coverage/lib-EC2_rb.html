 45   # Builds the canonical string for signing. This strips out all '&', '?', and '='
 46   # from the query string to be signed.
 47   #   Note:  The parameters in the path passed in must already be sorted in
 48   #   case-insensitive alphabetical order and must not be url encoded.
 49   def EC2.canonical_string(path)
 50     buf = ""
 51     path.split('&').each { |field|
 52       buf << field.gsub(/\&|\?/,"").sub(/=/,"")
 53     }
 54     return buf
 55   end

 57   # Encodes the given string with the aws_secret_access_key, by taking the
 58   # hmac-sha1 sum, and then base64 encoding it.  Optionally, it will also
 59   # url encode the result of that to protect the string if it's going to
 60   # be used as a query string parameter.
 61   def EC2.encode(aws_secret_access_key, str, urlencode=true)
 62     digest = OpenSSL::Digest::Digest.new('sha1')
 63     b64_hmac =
 64       Base64.encode64(

 66         
 67     if urlencode
 68       return CGI::escape(b64_hmac)

 70       return b64_hmac
 71     end
